📝 コード

const log = carbon.id('event-log');

carbon.id('touch-box')
    .tapped(() => log.text('💡 タップ！'))
    .longPressed(() => log.text('⏱️ 長押し！'), 500)
    .swiped((el, dir) => {
        log.text(`💨 ${dir}にスワイプ！`);
    });
