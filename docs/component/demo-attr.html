4. Attribute (Disable)

要素の属性操作です。Manipulate element attributes.

📝 コード📝 Code

let isDisabled = false;
const targetBtn = carbon.id('btn-target');

carbon.id('btn-disable').tapped((w) => {
    if (!isDisabled) {
        targetBtn.disable();
        w.text('有効化するEnable');
    } else {
        targetBtn.enable();
        w.text('無効化するDisable');
    }
    isDisabled = !isDisabled;
});
